TEST RANKING
-- Batters Ranking Table
CREATE TABLE TEST_BATTER_RANKINGS (
    rank INT PRIMARY KEY,
    player_id INT,
    player_name VARCHAR(100),
    country VARCHAR(50),
    rating DECIMAL(5,1),
    average DECIMAL(5,2),
    centuries INT,
    FOREIGN KEY (player_id) REFERENCES PLAYER(player_id)
);

-- Bowlers Ranking Table
CREATE TABLE TEST_BOWLER_RANKINGS (
    rank INT PRIMARY KEY,
    player_id INT,
    player_name VARCHAR(100),
    country VARCHAR(50),
    rating DECIMAL(5,1),
    average DECIMAL(5,2),
    wickets INT,
    FOREIGN KEY (player_id) REFERENCES PLAYER(player_id)
);

-- All-rounders Ranking Table
CREATE TABLE TEST_ALLROUNDER_RANKINGS (
    rank INT PRIMARY KEY,
    player_id INT,
    player_name VARCHAR(100),
    country VARCHAR(50),
    rating DECIMAL(5,1),
    batting_avg DECIMAL(5,2),
    bowling_avg DECIMAL(5,2),
    FOREIGN KEY (player_id) REFERENCES PLAYER(player_id)
);
